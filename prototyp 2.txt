<script>
const €=v=>v.toLocaleString("de-AT",{style:"currency",currency:"EUR"});
const $=id=>document.getElementById(id);
let page="setup",products=[],cart=[],paid=0,stats={revenue:0,products:{}};
const money=[0.05,0.10,0.20,0.50,1,2,5,10,20,50,100];

function load(){products=JSON.parse(localStorage.cashProducts||"[]");stats=JSON.parse(localStorage.cashStats||'{"revenue":0,"products":{}}')}
function save(){localStorage.cashProducts=JSON.stringify(products);localStorage.cashStats=JSON.stringify(stats)}
function notice(el,msg){el.textContent=msg;el.classList.add("show")}
function hide(el){el.classList.remove("show")}
function total(){return cart.reduce((s,i)=>s+i.price*i.qty,0)}
function show(p){page=p;["setup","sales","payment"].forEach(x=>$(x).classList.toggle("active",x===p));$("title").textContent={setup:"Produkte anlegen",sales:"Kassieren",payment:"Bezahlung"}[p];$("back").classList.toggle("show",p!=="setup");if(p==="sales")renderSales();if(p==="payment")renderPay()}

function addProduct(){
  let n=$("name").value.trim(), p=Number($("price").value.replace(",","."));
  if(!n)return notice($("setupNotice"),"Bitte gib einen Produktnamen ein.");
  if(!Number.isFinite(p)||p<=0)return notice($("setupNotice"),"Bitte gib einen gültigen Preis ein.");
  hide($("setupNotice"));products.push({id:Date.now()+"",name:n,price:Math.round(p*100)/100});
  save();$("name").value="";$("price").value="";renderProducts();
}
function renderProducts(){
  $("productList").innerHTML=products.length?"":"<p>Noch keine Produkte angelegt.</p>";
  products.forEach(p=>{
    let r=document.createElement("div");r.className="row";
    r.innerHTML=`<strong>${esc(p.name)}</strong><span>${€(p.price)}</span>`;
    let b=document.createElement("button");b.className="danger";b.textContent="Löschen";b.onclick=()=>{products=products.filter(x=>x.id!==p.id);cart=cart.filter(x=>x.id!==p.id);save();renderProducts()};
    r.appendChild(b);$("productList").appendChild(r);
  });
}
function renderSales(){
  $("productButtons").innerHTML="";
  products.forEach(p=>{
    let b=document.createElement("button");b.className="choice";b.innerHTML=`<strong>${esc(p.name)}</strong><span class="price">${€(p.price)}</span>`;
    b.onclick=()=>{let i=cart.find(x=>x.id===p.id);i?i.qty++:cart.push({...p,qty:1});renderCart()};
    $("productButtons").appendChild(b);
  });renderCart();
}
function renderCart(){
  $("cartList").innerHTML=cart.length?"":"<p>Der Einkauf ist noch leer.</p>";
  cart.forEach(i=>{
    let r=document.createElement("div");r.className="row";r.innerHTML=`<strong>${i.qty} × ${esc(i.name)}</strong><span>${€(i.price*i.qty)}</span>`;
    let c=document.createElement("div"),m=document.createElement("button"),p=document.createElement("button");
    m.className=p.className="light";m.textContent="-";p.textContent="+";
    m.onclick=()=>{i.qty--;if(i.qty<=0)cart=cart.filter(x=>x.id!==i.id);renderCart()};p.onclick=()=>{i.qty++;renderCart()};
    c.append(m,p);r.appendChild(c);$("cartList").appendChild(r);
  });
  if(cart.length){let r=document.createElement("div");r.className="row";r.innerHTML=`<strong>Gesamt</strong><strong>${€(total())}</strong><span></span>`;$("cartList").appendChild(r)}
}
function renderPay(){
  $("total").textContent=€(total());$("paid").textContent=€(paid);$("change").textContent=€(Math.max(0,paid-total()));
  paid>0&&paid<total()?notice($("payNotice"),"Der gegebene Betrag reicht noch nicht aus."):hide($("payNotice"));
  $("moneyButtons").innerHTML="";
  money.forEach(a=>{let b=document.createElement("button");b.className="choice";b.textContent=€(a);b.onclick=()=>{paid=Math.round((paid+a)*100)/100;renderPay()};$("moneyButtons").appendChild(b)});
  let exact=document.createElement("button");exact.className="choice blue";exact.textContent="Passend";exact.onclick=()=>{paid=total();renderPay()};
  let reset=document.createElement("button");reset.className="choice danger";reset.textContent="Geld löschen";reset.onclick=()=>{paid=0;renderPay()};
  $("moneyButtons").append(exact,reset);
}
function saveSale(){
  stats.revenue=Math.round((stats.revenue+total())*100)/100;
  cart.forEach(i=>{stats.products[i.id]??={name:i.name,quantity:0,revenue:0};stats.products[i.id].name=i.name;stats.products[i.id].quantity+=i.qty;stats.products[i.id].revenue=Math.round((stats.products[i.id].revenue+i.price*i.qty)*100)/100});
  save();
}
function renderStats(){
  let items=Object.values(stats.products),count=items.reduce((s,i)=>s+i.quantity,0);
  $("soldTotal").textContent=count;$("revenueTotal").textContent=€(stats.revenue);
  $("statsList").innerHTML=items.length?"":"<p>Noch keine Verkäufe abgeschlossen.</p>";
  items.sort((a,b)=>b.quantity-a.quantity).forEach(i=>{let r=document.createElement("div");r.className="row";r.innerHTML=`<strong>${esc(i.name)}</strong><span>${i.quantity} verkauft</span><strong>${€(i.revenue)}</strong>`;$("statsList").appendChild(r)});
}
function finish(){
  if(page==="setup"){if(!products.length)return notice($("setupNotice"),"Bitte lege zuerst mindestens ein Produkt an.");show("sales");return}
  if(page==="sales"){if(!cart.length)return;paid=0;show("payment");return}
  if(paid<total())return notice($("payNotice"),"Bitte wähle genug Geld aus, bevor du abschließt.");
  saveSale();cart=[];paid=0;show("sales");
}
function esc(t){let e=document.createElement("span");e.textContent=t;return e.innerHTML}

$("add").onclick=addProduct;$("finish").onclick=finish;$("back").onclick=()=>show(page==="payment"?"sales":"setup");
$("clearCart").onclick=()=>{cart=[];renderCart()};$("statsBtn").onclick=()=>{$("statsModal").classList.add("show");renderStats()};
$("closeStats").onclick=()=>$("statsModal").classList.remove("show");$("statsModal").onclick=e=>{if(e.target===$("statsModal"))$("statsModal").classList.remove("show")};
$("resetStats").onclick=()=>{stats={revenue:0,products:{}};save();renderStats()};
$("price").onkeydown=e=>{if(e.key==="Enter")addProduct()};$("name").onkeydown=e=>{if(e.key==="Enter")$("price").focus()};
load();renderProducts();
</script>
